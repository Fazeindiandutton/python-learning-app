Basic Math

Easy

Task: Create a program that calculates the area and perimeter of a rectangle.

Input: Width and height (float)

Output: Area and perimeter

Exercise 
# Rectangle Calculator
# Read dimensions
width = float(input("Enter width: "))
height = float(input("Enter height: "))

# Calculate area and perimeter
area = width * height
perimeter = 2 * (width + height)

# Display results
print(f"Area: {area}")
print(f"Perimeter: {perimeter}")

# Extra challenge: Check if it's a square
if width == height:
    print("It's a square!")
else:
    print("It's a rectangle!")

Number Operations

Medium

Task: Create a number guessing game where the user has 5 attempts to guess a random number.

Concept: Practice with loops and conditional logic

Exercise 
# Number Guessing Game
import random

# Generate random number
secret_number = random.randint(1, 100)
attempts = 5

print("Welcome to Number Guessing Game!")
print("You have", attempts, "attempts to guess the number.")

for attempt in range(attempts):
    guess = int(input(f"Attempt {attempt + 1}: Enter your guess: "))
    
    if guess == secret_number:
        print("Congratulations! You guessed the number!")
        break
    elif guess < secret_number:
        print("Too low!")
    else:
        print("Too high!")
else:
    print(f"Game over! The number was {secret_number}.")

Task: Create a simple shopping cart system that processes items and calculates totals.

Exercise 
# Shopping Cart
# Define items and prices
items = {
    "apple": 0.99,
    "banana": 0.50,
    "orange": 0.75,
    "bread": 2.49,
    "milk": 3.99
}

print("Available items:")
for item, price in items.items():
    print(f"  {item}: ${price:.2f}")

# Create shopping cart
cart = []
total = 0.0

while True:
    item = input("Enter item to add (or 'done' to checkout): ")
    if item.lower() == 'done':
        break
    
    if item in items:
        cart.append(item)
        total += items[item]
        print(f"Added {item} to cart.")
    else:
        print("Item not available.")

# Display receipt
print("\nShopping Cart:")
for item in cart:
    print(f"  - {item}: ${items[item]:.2f}")

print(f"\nTotal: ${total:.2f}")

# Add tax (assume 8% sales tax)
tax_rate = 0.08
final_total = total * (1 + tax_rate)
print(f"Total with tax (8%): ${final_total:.2f}")

Medium

Task: Create a simple student grade management system.

Exercise 
# Student Grade Management
# Create data structure
students = {
    "Alice": [85, 92, 78, 95],
    "Bob": [72, 88, 91, 84],
    "Charlie": [95, 87, 92, 89],
    "Diana": [65, 72, 68, 75]
}

# Calculate statistics
print("Student Grades Analysis:")
print("=" * 40)

for name, grades in students.items():
    total = sum(grades)
    average = total / len(grades)
    highest = max(grades)
    lowest = min(grades)
    
    print(f"{name}: {grades}")
    print(f"  Average: {average:.1f}")
    print(f"  Highest: {highest}, Lowest: {lowest}")
    
    # Letter grade
    if average >= 90:
        letter_grade = "A"
    elif average >= 80:
        letter_grade = "B"
    elif average >= 70:
        letter_grade = "C"
    else:
        letter_grade = "D"
    
    print(f"  Grade: {letter_grade}")
    print()

# Find top student
top_student = max(students.keys(), key=lambda name: sum(students[name])/len(students[name]))
print(f"Top student: {top_student}")

# Calculate overall class statistics
all_grades = []
for grades in students.values():
    all_grades.extend(grades)

class_average = sum(all_grades) / len(all_grades)
print(f"Class average: {class_average:.1f}")
print(f"Class total grades: {len(all_grades)}")

Hard

Task: Create a text-based adventure game with simple choices.

Exercise 
# Simple Text Adventure Game

def print_intro():
    print("Welcome to the Treasure Cave!")
    print("You find yourself standing at the entrance of a dark cave.")
    print("There's a glimmer of light coming from inside.")
    print("\nWhat do you do?")
    print("1. Enter the cave")
    print("2. Look for another way in")
    print("3. Set up camp and wait")

def explore_cave():
    print("\nYou carefully enter the dark cave...")
    print("After 10 minutes, you find:")
    print("1. A shiny treasure chest")
    print("2. A sleeping bear")
    print("3. A dead end")
    
    choice = input("Choose 1, 2, or 3: ")
    
    if choice == '1':
        print("\nYou open the treasure chest and find gold! You win!")
    elif choice == '2':
        print("\nThe bear wakes up and runs away! You escape with some valuables.")
    else:
        print("\nIt's a dead end. You find your way back.")

def find_other_way():
    print("\nYou look for another way into the cave...")
    print("You find a narrow tunnel that leads to...")
    print("1. A beautiful garden")
    print("2. A dark dungeon")
    print("3. The treasure chamber")
    
    choice = input("Choose 1, 2, or 3: ")
    
    if choice == '1':
        print("\nYou find a beautiful garden with rare herbs!")
    elif choice == '2':
        print("\nYou stumble into a trap and barely escape!")
    else:
        print("\nYou find the treasure chamber but it's empty.")

def wait_and_rest():
    print("\nYou set up camp and wait...")
    print("Suddenly, you hear mysterious footsteps...")
    print("1. Hide and observe")
    print("2. Approach cautiously")
    print("3. Run back to camp")
    
    choice = input("Choose 1, 2, or 3: ")
    
    if choice == '1':
        print("\nYou hide and watch as friendly travelers pass by and give you supplies.")
    elif choice == '2':
        print("\nYou approach and befriend the travelers who help you explore.")
    else:
        print("\nYou flee in panic but lose valuable time.")

# Main game loop
print_intro()
choice = input("Choose 1, 2, or 3: ")

if choice == '1':
    explore_cave()
elif choice == '2':
    find_other_way()
elif choice == '3':
    wait_and_rest()
else:
    print("\nInvalid choice. The darkness claims you...")

print("\nGame Over! Thanks for playing!")
